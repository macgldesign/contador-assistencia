        function updateUI() {
            // Limpa a seleÃ§Ã£o visual de todos os assentos.
            document.querySelectorAll('.seat.selected').forEach(seat => {
                seat.classList.remove('selected');

            });

            // Re-seleciona os assentos com base no estado carregado.

        // Cria a grade de assentos para uma seÃ§Ã£o especÃ­fica.
        function createSeats(sectionId, layout) {
            const section = document.getElementById(sectionId);
            section.innerHTML = ''; // Limpa a grade existente
            const { rows, cols } = layout;
            for (let i = 0; i < rows; i++) {
                const row = document.createElement('div');
                row.className = 'seat-row';
                for (let j = 0; j < cols; j++) {
                    const seat = document.createElement('div');
                    seat.className = 'seat';
                    seat.onclick = toggleSeat;
                    row.appendChild(seat);
                }
                section.appendChild(row);
            }
        }

        function recreateAllSeats() {
            createSeats('left-seats', seatLayout.left);
            createSeats('center-seats', seatLayout.center);

        }

        // --- 3. FunÃ§Ãµes de PersistÃªncia de Dados ---

        // Salva o estado atual da aplicaÃ§Ã£o no localStorage.
        function saveState() {
            try {
                localStorage.setItem(storageKey, JSON.stringify(appState));
                localStorage.setItem(layoutStorageKey, JSON.stringify(seatLayout));
            } catch (e) {
                console.error("Erro ao salvar estado no localStorage:", e);
            }
        }
        
        // Carrega o estado do localStorage e o aplica, se disponÃ­vel.
        function loadState() {
            try {
                const savedState = localStorage.getItem(storageKey);
                if (savedState) {
                    const parsedState = JSON.parse(savedState);
                    if (parsedState && parsedState.seats && typeof parsedState.standCount === 'number') {
                        appState = parsedState;
                    } else {
                        console.warn("Dados corrompidos no localStorage. O estado de assistÃªncia serÃ¡ redefinido.");
                        localStorage.removeItem(storageKey);
                    }
                }
                const savedLayout = localStorage.getItem(layoutStorageKey);
                if (savedLayout) {
                    const parsedLayout = JSON.parse(savedLayout);
                    // Atualiza apenas os layouts que foram salvos
                    if (parsedLayout.left) seatLayout.left = parsedLayout.left;
                    if (parsedLayout.center) seatLayout.center = parsedLayout.center;
                    if (parsedLayout.right) seatLayout.right = parsedLayout.right;
                    if (parsedLayout.salaB) seatLayout.salaB = parsedLayout.salaB;
                }
            } catch (e) {
                console.error("Erro ao carregar estado do localStorage:", e);
                localStorage.removeItem(storageKey);
                localStorage.removeItem(layoutStorageKey);
            }
        }

        // --- 4. Manipuladores de Eventos ---

        // Alterna o estado de seleÃ§Ã£o de um assento.
        function toggleSeat(event) {
            const seat = event.target.closest('.seat');
            if (!seat) return;
            
            const isSelected = seat.classList.contains('selected');
            const sectionElement = seat.closest('.seat-section');
            const sectionTitle = sectionElement.querySelector('h2').textContent.trim().toLowerCase();
            let sectionId;
            
            if (sectionTitle === 'esquerda') {
                sectionId = 'left';
            } else if (sectionTitle === 'central') {
                sectionId = 'center';
            } else if (sectionTitle === 'direita') {
                sectionId = 'right';
            } else if (sectionTitle === 'sala b') {
                sectionId = 'salaB';
            }

            if (!sectionId) return;

            const seatRowIndex = Array.from(seat.parentNode.parentNode.children).indexOf(seat.parentNode);
            const seatColIndex = Array.from(seat.parentNode.children).indexOf(seat);
            
            if (isSelected) {
                appState.seats[sectionId] = appState.seats[sectionId].filter(
                    s => s.row !== seatRowIndex || s.col !== seatColIndex
                );
            } else {
                appState.seats[sectionId].push({ row: seatRowIndex, col: seatColIndex });
            }
            
            updateUI();
            saveState();
        }

        // Zera todos os contadores e limpa o armazenamento.

            appState.seats = {
                left: [],
                center: [],
                right: [],
                salaB: []
            };
            appState.standCount = 0;
            updateUI();
            localStorage.removeItem(storageKey);
        }

        function openSettingsModal(sectionId) {
            currentSectionId = sectionId;
            const { rows, cols } = seatLayout[sectionId];
            document.getElementById('modal-title').textContent = `ConfiguraÃ§Ãµes - ${sectionId.toUpperCase()}`;
            document.getElementById('rows-count').textContent = rows;
            document.getElementById('cols-count').textContent = cols;
            document.getElementById('settings-modal').classList.add('show');
        }

        function closeSettingsModal() {
            document.getElementById('settings-modal').classList.remove('show');
            currentSectionId = null;
        }

        function updateLayout(type, action) {
            if (!currentSectionId) return;

            const layout = seatLayout[currentSectionId];
            if (type === 'rows') {
                if (action === 'add') {
                    layout.rows++;
                } else if (action === 'remove' && layout.rows > 1) {
                    layout.rows--;
                }
                document.getElementById('rows-count').textContent = layout.rows;
            } else if (type === 'cols') {
                if (action === 'add') {
                    layout.cols++;
                } else if (action === 'remove' && layout.cols > 1) {
                    layout.cols--;
                }
                document.getElementById('cols-count').textContent = layout.cols;
            }

            // Trunca assentos selecionados se o layout encolher
            appState.seats[currentSectionId] = appState.seats[currentSectionId].filter(
                seat => seat.row < layout.rows && seat.col < layout.cols
            );
            
            recreateAllSeats();
            updateUI();
            saveState();
        }
