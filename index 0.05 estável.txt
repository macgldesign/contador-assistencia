    <script>
        // --- 1. Definições de Estado e Layout ---
        
        // Define o layout de assentos para cada seção do auditório.
        const seatLayout = {
            left: [
                { rows: 9, cols: 4 }
            ],
            center: [
                { rows: 9, cols: 8 }
            ],
            right: [
                { rows: 9, cols: 4 }
            ],
            salaB: [
                { rows: 5, cols: 5 }
            ]
        };
        const storageKey = 'seatCounterState';
        let isAdvancedShareMode = false;
        
        // Objeto de estado central para gerenciar todos os dados da aplicação.
        let appState = {
            seats: {
                left: [],
                center: [],
                right: [],
                salaB: []
            },
            standCount: 0
        };

        // --- 2. Funções de Gerenciamento da UI ---

        // Ajusta o preenchimento do conteúdo para evitar que o cabeçalho flutuante o cubra.
        function adjustContentPadding() {
            const header = document.getElementById('main-header');
            const content = document.getElementById('main-content');
            if (header && content) {
                content.style.marginTop = `${header.offsetHeight + 10}px`;
            }
        }
        
        // Atualiza a interface do usuário com base nos dados do appState.
        function updateUI() {
            // Limpa a seleção visual de todos os assentos.
            document.querySelectorAll('.seat.selected').forEach(seat => {
                seat.classList.remove('selected');
                seat.innerHTML = '';
            });

            // Re-seleciona os assentos com base no estado carregado.
            Object.keys(appState.seats).forEach(sectionId => {
                const sectionElement = document.getElementById(`${sectionId.toLowerCase().replace('b', '-b')}-seats`);
                
                if (sectionElement) {
                    appState.seats[sectionId].forEach(({ row, col }) => {
                        const seatRow = sectionElement.children[row];
                        if (seatRow) {
                            const seat = seatRow.children[col];
                            if (seat) {
                                seat.classList.add('selected');
                                seat.innerHTML = `<svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5 text-white" viewBox="0 0 20 20" fill="currentColor">
                                    <path fill-rule="evenodd" d="M16.707 5.293a1 1 0 010 1.414l-8 8a1 1 0 01-1.414 0l-4-4a1 1 0 011.414-1.414L8 12.586l7.293-7.293a1 1 0 011.414 0z" clip-rule="evenodd" />
                                </svg>`;
                            }
                        }
                    });
                }
            });

            // Atualiza os contadores na UI.
            document.getElementById('stand-count-input').value = appState.standCount;
            const leftSeats = appState.seats.left.length;
            const centerSeats = appState.seats.center.length;
            const rightSeats = appState.seats.right.length;
            const salaBSeats = appState.seats.salaB.length;
            const total = leftSeats + centerSeats + rightSeats + salaBSeats + appState.standCount;

            document.getElementById('total-count').textContent = total;
            document.getElementById('left-count').textContent = leftSeats;
            document.getElementById('center-count').textContent = centerSeats;
            document.getElementById('right-count').textContent = rightSeats;
            document.getElementById('sala-b-count').textContent = salaBSeats;

            const pessoasText = total === 1 ? "PESSOA" : "PESSOAS";
            document.querySelector('.summary-box p').innerHTML = `ASSISTÊNCIA: <span id="total-count">${total}</span> ${pessoasText}`;

            // Controla a ativação dos botões e estilos.
            const shareButton = document.getElementById('share-button');
            const resetButton = document.getElementById('reset-button');
            const summaryBox = document.querySelector('.summary-box');

            if (total === 0) {
                shareButton.disabled = true;
                resetButton.disabled = true;
                summaryBox.classList.add('zero-count');
                summaryBox.classList.remove('advanced-share-mode');
                isAdvancedShareMode = false;
            } else {
                shareButton.disabled = false;
                resetButton.disabled = false;
                summaryBox.classList.remove('zero-count');
            }
        }
        
        // Cria a grade de assentos para uma seção específica.
        function createSeats(sectionId, layout) {
            const section = document.getElementById(sectionId);
            layout.forEach(group => {
                for (let i = 0; i < group.rows; i++) {
                    const row = document.createElement('div');
                    row.className = 'seat-row';
                    for (let j = 0; j < group.cols; j++) {
                        const seat = document.createElement('div');
                        seat.className = 'seat';
                        seat.onclick = toggleSeat;
                        row.appendChild(seat);
                    }
                    section.appendChild(row);
                }
            });
        }

        // --- 3. Funções de Persistência de Dados ---

        // Salva o estado atual da aplicação no localStorage.
        function saveState() {
            try {
                localStorage.setItem(storageKey, JSON.stringify(appState));
            } catch (e) {
                console.error("Erro ao salvar estado no localStorage:", e);
            }
        }
        
        // Carrega o estado do localStorage e o aplica, se disponível.
        function loadState() {
            try {
                const savedState = localStorage.getItem(storageKey);
                if (savedState) {
                    const parsedState = JSON.parse(savedState);
                    if (parsedState && parsedState.seats && typeof parsedState.standCount === 'number') {
                        appState = parsedState;
                    } else {
                        console.warn("Dados corrompidos no localStorage. O estado será redefinido.");
                        localStorage.removeItem(storageKey);
                    }
                }
            } catch (e) {
                console.error("Erro ao carregar estado do localStorage:", e);
                localStorage.removeItem(storageKey);
            }
        }

        // --- 4. Manipuladores de Eventos ---

        // Alterna o estado de seleção de um assento.
        function toggleSeat(event) {
            const seat = event.target.closest('.seat');
            if (!seat) return;
            
            const isSelected = seat.classList.contains('selected');
            const sectionElement = seat.closest('.seat-section');
            const sectionTitle = sectionElement.querySelector('h2').textContent.trim().toLowerCase();
            let sectionId;
            
            if (sectionTitle === 'esquerda') {
                sectionId = 'left';
            } else if (sectionTitle === 'central') {
                sectionId = 'center';
            } else if (sectionTitle === 'direita') {
                sectionId = 'right';
            } else if (sectionTitle === 'sala b') {
                sectionId = 'salaB';
            }

            if (!sectionId) return;

            const seatRowIndex = Array.from(seat.parentNode.parentNode.children).indexOf(seat.parentNode);
            const seatColIndex = Array.from(seat.parentNode.children).indexOf(seat);
            
            if (isSelected) {
                appState.seats[sectionId] = appState.seats[sectionId].filter(
                    s => s.row !== seatRowIndex || s.col !== seatColIndex
                );
            } else {
                appState.seats[sectionId].push({ row: seatRowIndex, col: seatColIndex });
            }
            
            updateUI();
            saveState();
        }

        // Zera todos os contadores e limpa o armazenamento.
        function resetAll() {
            appState.seats = {
                left: [],
                center: [],
                right: [],
                salaB: []
            };
            appState.standCount = 0;
            updateUI();
            localStorage.removeItem(storageKey);
        }
        
        // Gera e compartilha a mensagem de contagem no WhatsApp.
        function shareOnWhatsApp() {
            const today = new Date();
            const day = String(today.getDate()).padStart(2, '0');
            const month = String(today.getMonth() + 1).padStart(2, '0');
            const year = today.getFullYear();
            const formattedDate = `${day}/${month}/${year}`;
            
            let message;
            if (isAdvancedShareMode) {
                const total = appState.seats.left.length + appState.seats.center.length + appState.seats.right.length + appState.seats.salaB.length + appState.standCount;
                message = `Assistência presencial ${formattedDate}:\nEsquerda: ${appState.seats.left.length}\nCentral: ${appState.seats.center.length}\nDireita: ${appState.seats.right.length}\nSala B: ${appState.seats.salaB.length}\nEm pé: ${appState.standCount}\n*Total: ${total} pessoas*`;
            } else {
                const total = parseInt(document.querySelector('#total-count').textContent, 10);
                const pessoasText = total === 1 ? "pessoa" : "pessoas";
                message = `${formattedDate}\nAssistência presencial: *${total} ${pessoasText}*`;
            }

            const whatsappUrl = `https://api.whatsapp.com/send?text=${encodeURIComponent(message)}`;
            window.open(whatsappUrl, '_blank');
        }

        // --- 5. Inicialização da Aplicação ---

        // Garante que o código seja executado somente após o carregamento completo do DOM.
        document.addEventListener('DOMContentLoaded', () => {
            // Carrega o estado salvo antes de construir a UI.
            loadState();

            // Cria os assentos na interface na ordem de exibição.
            createSeats('left-seats', seatLayout.left);
            createSeats('center-seats', seatLayout.center);
            createSeats('right-seats', seatLayout.right);
            createSeats('sala-b-seats', seatLayout.salaB);
            
            // Atualiza a UI com o estado carregado.
            updateUI();

            // Adiciona ouvintes de eventos aos botões.
            document.getElementById('reset-button').onclick = resetAll;
            document.getElementById('share-button').onclick = shareOnWhatsApp;

            document.getElementById('summary-box').ondblclick = () => {
                const total = appState.seats.left.length + appState.seats.center.length + appState.seats.right.length + appState.seats.salaB.length + appState.standCount;
                if (total > 0) {
                    isAdvancedShareMode = !isAdvancedShareMode;
                    const summaryBox = document.getElementById('summary-box');
                    if (isAdvancedShareMode) {
                        summaryBox.classList.add('advanced-share-mode');
                    } else {
                        summaryBox.classList.remove('advanced-share-mode');
                    }
                }
            };

            const standInput = document.getElementById('stand-count-input');
            const plusButton = document.getElementById('plus-button');
            const minusButton = document.getElementById('minus-button');
            
            function handleStandCount(isPlus) {
                if (isPlus) {
                    appState.standCount++;
                } else {
                    if (appState.standCount > 0) {
                        appState.standCount--;
                    }
                }
                updateUI();
                saveState();
            }

            plusButton.addEventListener('click', () => handleStandCount(true));
            plusButton.addEventListener('touchstart', (e) => {
                e.preventDefault();
                handleStandCount(true);
            });

            minusButton.addEventListener('click', () => handleStandCount(false));
            minusButton.addEventListener('touchstart', (e) => {
                e.preventDefault();
                handleStandCount(false);
            });
            
            standInput.oninput = (e) => {
                const value = parseInt(e.target.value, 10);
                appState.standCount = Math.max(0, isNaN(value) ? 0 : value);
                updateUI();
                saveState();
            };
            
            adjustContentPadding();
        });
    </script>
